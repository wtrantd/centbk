### Sharing my project repo on public ###

# install git
yum install -y git

# user global configuration
git config --global user.name "abc"
git config --global user.email "abc@domain.com"

# editor
git config --system core.editor "/bin/vim"

# Creating a local repository and adding content 

mkdir ~/repo
git init ~/repo
ls -al ~/repo
