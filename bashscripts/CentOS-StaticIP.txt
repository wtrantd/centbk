#!/bin/bash
#
# This script is used for setting up an static ip address

cp /etc/sysconfig/network-scripts/ifcfg-ens34 /etc/sysconfig/network-scripts/ifcfg-ens34.bk
sed -i -e s/dhcp/static/ /etc/sysconfig/network-scripts/ifcfg-ens34 
sed -i -e s/ONBOOT=no/ONBOOT=yes/ /etc/sysconfig/network-scripts/ifcfg-ens34
echo "IPADDR=192.168.1.21
PREFIX=24
GATEWAY=192.168.1.2
DNS1=192.168.1.2" >> /etc/sysconfig/network-scripts/ifcfg-ens34
systemctl restart network
ip a s | grep inet
ping -c 2 google.com || echo ping google fails
exit 0
